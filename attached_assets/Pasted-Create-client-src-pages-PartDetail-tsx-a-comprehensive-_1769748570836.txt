async getPartProcessMappings(partId: string) {
  return db
    .select({
      id: partProcessMap.id,
      processDefId: partProcessMap.processDefId,
      processName: processDef.name,
      processRev: partProcessMap.processRev,
      sequence: partProcessMap.sequence,
    })
    .from(partProcessMap)
    .leftJoin(processDef, eq(partProcessMap.processDefId, processDef.id))
    .where(eq(partProcessMap.partId, partId))
    .orderBy(partProcessMap.sequence);
}
