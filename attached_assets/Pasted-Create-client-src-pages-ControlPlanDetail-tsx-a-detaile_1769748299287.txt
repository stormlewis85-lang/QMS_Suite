export default function ControlPlanDetail() {
  const { id } = useParams<{ id: string }>();
  const [, navigate] = useLocation();
  const { toast } = useToast();
  const queryClient = useQueryClient();
  
  // State
  const [filterType, setFilterType] = useState<string>('all');
  const [filterSpecial, setFilterSpecial] = useState<boolean>(false);
  const [searchTerm, setSearchTerm] = useState('');
  const [editingRow, setEditingRow] = useState<ControlPlanRow | null>(null);
  const [editFormData, setEditFormData] = useState<Partial<ControlPlanRow>>({});
  const [isSaving, setIsSaving] = useState(false);
  
  // Queries
  const { data: controlPlan, isLoading: cpLoading } = useQuery({
    queryKey: [`/api/control-plans/${id}`],
  });
  
  const { data: rows, isLoading: rowsLoading } = useQuery({
    queryKey: [`/api/control-plans/${id}/rows`],
  });
  
  // Computed values
  const summary = useMemo(() => {
    if (!rows) return { total: 0, product: 0, process: 0, special: 0, withTarget: 0 };
    return {
      total: rows.length,
      product: rows.filter(r => r.type === 'Product').length,
      process: rows.filter(r => r.type === 'Process').length,
      special: rows.filter(r => r.specialFlag).length,
      withTarget: rows.filter(r => r.target).length,
    };
  }, [rows]);
  
  const filteredRows = useMemo(() => {
    if (!rows) return [];
    return rows.filter(row => {
      if (filterType !== 'all' && row.type !== filterType) return false;
      if (filterSpecial && !row.specialFlag) return false;
      if (searchTerm) {
        const search = searchTerm.toLowerCase();
        return (
          row.characteristicName.toLowerCase().includes(search) ||
          row.charId.toLowerCase().includes(search) ||
          row.controlMethod?.toLowerCase().includes(search) ||
          row.measurementSystem?.toLowerCase().includes(search)
        );
      }
      return true;
    });
  }, [rows, filterType, filterSpecial, searchTerm]);
  
  // Mutations
  const updateRowMutation = useMutation({
    mutationFn: async (data: { rowId: string; updates: Partial<ControlPlanRow> }) => {
      const response = await fetch(`/api/control-plan-rows/${data.rowId}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data.updates),
      });
      if (!response.ok) {
        const error = await response.json();
        throw new Error(error.error || 'Failed to update row');
      }
      return response.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: [`/api/control-plans/${id}/rows`] });
      toast({
        title: "Row Updated",
        description: "Control Plan row has been updated successfully.",
      });
      setEditingRow(null);
      setEditFormData({});
    },
    onError: (error: Error) => {
      toast({
        title: "Update Failed",
        description: error.message,
        variant: "destructive",
      });
    },
  });
  
  // Handlers
  const handleEditClick = (row: ControlPlanRow) => {
    setEditingRow(row);
    setEditFormData({
      characteristicName: row.characteristicName,
      type: row.type,
      target: row.target,
      tolerance: row.tolerance,
      specialFlag: row.specialFlag,
      csrSymbol: row.csrSymbol,
      measurementSystem: row.measurementSystem,
      gageDetails: row.gageDetails,
      sampleSize: row.sampleSize,
      frequency: row.frequency,
      controlMethod: row.controlMethod,
      acceptanceCriteria: row.acceptanceCriteria,
      reactionPlan: row.reactionPlan,
    });
  };
  
  const handleSaveEdit = async () => {
    if (!editingRow) return;
    setIsSaving(true);
    
    // Track which fields were overridden from template
    const overrideFlags: Record<string, boolean> = { ...editingRow.overrideFlags };
    const templateFields = ['target', 'tolerance', 'sampleSize', 'frequency', 'controlMethod', 'acceptanceCriteria', 'reactionPlan'];
    templateFields.forEach(field => {
      if (editFormData[field as keyof typeof editFormData] !== editingRow[field as keyof typeof editingRow]) {
        overrideFlags[field] = true;
      }
    });
    
    await updateRowMutation.mutateAsync({
      rowId: editingRow.id,
      updates: {
        ...editFormData,
        overrideFlags,
      },
    });
    
    setIsSaving(false);
  };
  
  const handleCancelEdit = () => {
    setEditingRow(null);
    setEditFormData({});
  };
  
  // Loading state
  if (cpLoading || rowsLoading) {
    return (
      <div className="container mx-auto p-6 space-y-6">
        <Skeleton className="h-8 w-64" />
        <Skeleton className="h-32 w-full" />
        <Skeleton className="h-96 w-full" />
      </div>
    );
  }
  
  if (!controlPlan) {
    return (
      <div className="container mx-auto p-6">
        <Alert variant="destructive">
          <AlertTriangle className="h-4 w-4" />
          <AlertDescription>Control Plan not found</AlertDescription>
        </Alert>
        <Button variant="outline" className="mt-4" onClick={() => navigate('/control-plans')}>
          <ArrowLeft className="h-4 w-4 mr-2" /> Back to Control Plans
        </Button>
      </div>
    );
  }
  
  return (
    <div className="container mx-auto p-6 space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-4">
          <Button variant="ghost" size="sm" onClick={() => navigate('/control-plans')}>
            <ArrowLeft className="h-4 w-4 mr-1" /> Back
          </Button>
          <div>
            <h1 className="text-2xl font-bold flex items-center gap-2">
              <ClipboardList className="h-6 w-6" />
              Control Plan: {controlPlan.part?.partNumber || 'Unknown Part'}
            </h1>
            <p className="text-muted-foreground">
              {controlPlan.part?.partName} | Rev {controlPlan.rev} | {controlPlan.type}
            </p>
          </div>
        </div>
        <div className="flex items-center gap-2">
          <Badge variant={
            controlPlan.status === 'effective' ? 'default' :
            controlPlan.status === 'draft' ? 'secondary' :
            controlPlan.status === 'review' ? 'outline' : 'destructive'
          }>
            {controlPlan.status}
          </Badge>
          <Badge variant="outline">{controlPlan.type}</Badge>
          <Button variant="outline" size="sm">
            <Download className="h-4 w-4 mr-1" /> Export
          </Button>
        </div>
      </div>
      
      {/* Summary Cards */}
      <div className="grid grid-cols-5 gap-4">
        <Card>
          <CardHeader className="pb-2">
            <CardDescription>Total Characteristics</CardDescription>
            <CardTitle className="text-2xl">{summary.total}</CardTitle>
          </CardHeader>
        </Card>
        <Card className="border-blue-200 bg-blue-50">
          <CardHeader className="pb-2">
            <CardDescription>Product</CardDescription>
            <CardTitle className="text-2xl text-blue-600">{summary.product}</CardTitle>
          </CardHeader>
        </Card>
        <Card className="border-cyan-200 bg-cyan-50">
          <CardHeader className="pb-2">
            <CardDescription>Process</CardDescription>
            <CardTitle className="text-2xl text-cyan-600">{summary.process}</CardTitle>
          </CardHeader>
        </Card>
        <Card className="border-purple-200 bg-purple-50">
          <CardHeader className="pb-2">
            <CardDescription>Special Chars</CardDescription>
            <CardTitle className="text-2xl text-purple-600">{summary.special}</CardTitle>
          </CardHeader>
        </Card>
        <Card className="border-green-200 bg-green-50">
          <CardHeader className="pb-2">
            <CardDescription>With Targets</CardDescription>
            <CardTitle className="text-2xl text-green-600">{summary.withTarget}</CardTitle>
          </CardHeader>
        </Card>
      </div>
      
      {/* Filters */}
      <Card>
        <CardContent className="pt-4">
          <div className="flex items-center gap-4">
            <div className="flex items-center gap-2">
              <Filter className="h-4 w-4 text-muted-foreground" />
              <span className="text-sm font-medium">Filters:</span>
            </div>
            <Select value={filterType} onValueChange={setFilterType}>
              <SelectTrigger className="w-36">
                <SelectValue placeholder="Type" />

            </Select>
            <div className="flex items-center gap-2">

              <Label htmlFor="filterSpecial" className="text-sm">Special Only</Label>

              <TableBody>
                {filteredRows.map((row) => (
                  <TableRow
