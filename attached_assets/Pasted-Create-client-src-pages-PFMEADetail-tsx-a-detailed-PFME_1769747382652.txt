interface PFMEARow {
  id: string;
  pfmeaId: string;
  parentTemplateRowId: string | null;
  stepRef: string;
  function: string;
  requirement: string;
  failureMode: string;
  effect: string;
  severity: number;
  cause: string;
  occurrence: number;
  preventionControls: string[];
  detectionControls: string[];
  detection: number;
  ap: 'H' | 'M' | 'L';
  specialFlag: boolean;
  csrSymbol: string | null;
  notes: string | null;
  overrideFlags: Record<string, boolean>;
}

interface PFMEA {
  id: string;
  partId: string;
  rev: string;
  status: 'draft' | 'review' | 'effective' | 'superseded' | 'obsolete';
  basis: string;
  docNo: string | null;
  approvedBy: string | null;
  approvedAt: string | null;
  effectiveFrom: string | null;
  part?: { partNumber: string; partName: string; customer: string; program: string; };
}

COMPONENT STRUCTURE:
export default function PFMEADetail() {
  const { id } = useParams<{ id: string }>();
  const [, navigate] = useLocation();
  const { toast } = useToast();
  const queryClient = useQueryClient();
  
  // State
  const [filterAP, setFilterAP] = useState<string>('all');
  const [filterStep, setFilterStep] = useState<string>('all');
  const [searchTerm, setSearchTerm] = useState('');
  const [editingRowId, setEditingRowId] = useState<string | null>(null);
  const [expandedRows, setExpandedRows] = useState<Set<string>>(new Set());
  
  // Queries
  const { data: pfmea, isLoading: pfmeaLoading } = useQuery({
    queryKey: [`/api/pfmeas/${id}`],
  });
  
  const { data: rows, isLoading: rowsLoading } = useQuery({
    queryKey: [`/api/pfmeas/${id}/rows`],
  });
  
  // Computed values
  const summary = useMemo(() => {
    if (!rows) return { total: 0, high: 0, medium: 0, low: 0, special: 0 };
    return {
      total: rows.length,
      high: rows.filter(r => r.ap === 'H').length,
      medium: rows.filter(r => r.ap === 'M').length,
      low: rows.filter(r => r.ap === 'L').length,
      special: rows.filter(r => r.specialFlag).length,
    };
  }, [rows]);
  
  const uniqueSteps = useMemo(() => {
    if (!rows) return [];
    return [...new Set(rows.map(r => r.stepRef))];
  }, [rows]);
  
  const filteredRows = useMemo(() => {
    if (!rows) return [];
    return rows.filter(row => {
      if (filterAP !== 'all' && row.ap !== filterAP) return false;
      if (filterStep !== 'all' && row.stepRef !== filterStep) return false;
      if (searchTerm) {
        const search = searchTerm.toLowerCase();
        return (
          row.failureMode.toLowerCase().includes(search) ||
          row.effect.toLowerCase().includes(search) ||
          row.cause.toLowerCase().includes(search) ||
          row.function.toLowerCase().includes(search)
        );
      }
      return true;
    });
  }, [rows, filterAP, filterStep, searchTerm]);
  
  // Group rows by step for better visualization
  const rowsByStep = useMemo(() => {
    const grouped: Record<string, PFMEARow[]> = {};
    filteredRows.forEach(row => {
      if (!grouped[row.stepRef]) grouped[row.stepRef] = [];
      grouped[row.stepRef].push(row);
    });
    return grouped;
  }, [filteredRows]);

  // Loading state
  if (pfmeaLoading || rowsLoading) {
    return (
      <div className="container mx-auto p-6 space-y-6">
        <Skeleton className="h-8 w-64" />
        <Skeleton className="h-32 w-full" />
        <Skeleton className="h-96 w-full" />
      </div>
    );
  }

  if (!pfmea) {
    return (
      <div className="container mx-auto p-6">
        <Alert variant="destructive">
          <AlertTriangle className="h-4 w-4" />
          <AlertDescription>PFMEA not found</AlertDescription>
        </Alert>
        <Button variant="outline" className="mt-4" onClick={() => navigate('/pfmea')}>
          <ArrowLeft className="h-4 w-4 mr-2" /> Back to PFMEAs
        </Button>
      </div>
    );
  }

  return (
    <div className="container mx-auto p-6 space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-4">
          <Button variant="ghost" size="sm" onClick={() => navigate('/pfmea')}>
            <ArrowLeft className="h-4 w-4 mr-1" /> Back
          </Button>
          <div>
            <h1 className="text-2xl font-bold flex items-center gap-2">
              <FileText className="h-6 w-6" />
              PFMEA: {pfmea.part?.partNumber || 'Unknown Part'}
            </h1>
            <p className="text-muted-foreground">
              {pfmea.part?.partName} | Rev {pfmea.rev} | {pfmea.basis}
            </p>
          </div>
        </div>
        <div className="flex items-center gap-2">
          <Badge variant={
            pfmea.status === 'effective' ? 'default' :
            pfmea.status === 'draft' ? 'secondary' :
            pfmea.status === 'review' ? 'outline' : 'destructive'
          }>
            {pfmea.status}
          </Badge>
          <Button variant="outline" size="sm">
            <Download className="h-4 w-4 mr-1" /> Export
          </Button>
        </div>
      </div>
      
      {/* Summary Cards */}
      <div className="grid grid-cols-5 gap-4">
        <Card>
          <CardHeader className="pb-2">
            <CardDescription>Total Rows</CardDescription>
            <CardTitle className="text-2xl">{summary.total}</CardTitle>
          </CardHeader>
        </Card>
        <Card className="border-red-200 bg-red-50">
          <CardHeader className="pb-2">
            <CardDescription>High AP</CardDescription>
            <CardTitle className="text-2xl text-red-600">{summary.high}</CardTitle>
          </CardHeader>
        </Card>
        <Card className="border-yellow-200 bg-yellow-50">
          <CardHeader className="pb-2">
            <CardDescription>Medium AP</CardDescription>
            <CardTitle className="text-2xl text-yellow-600">{summary.medium}</CardTitle>
          </CardHeader>
        </Card>
        <Card className="border-green-200 bg-green-50">
          <CardHeader className="pb-2">
            <CardDescription>Low AP</CardDescription>
            <CardTitle className="text-2xl text-green-600">{summary.low}</CardTitle>
          </CardHeader>
        </Card>
        <Card className="border-purple-200 bg-purple-50">
          <CardHeader className="pb-2">
            <CardDescription>Special Chars</CardDescription>
            <CardTitle className="text-2xl text-purple-600">{summary.special}</CardTitle>
          </CardHeader>
        </Card>
      </div>
      
      {/* Filters */}
      <Card>
        <CardContent className="pt-4">
          <div className="flex items-center gap-4">
            <div className="flex items-center gap-2">
              <Filter className="h-4 w-4 text-muted-foreground" />
              <span className="text-sm font-medium">Filters:</span>
            </div>
            <Select value={filterAP} onValueChange={setFilterAP}>
              <SelectTrigger className="w-32">
                <SelectValue placeholder="AP Level" />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">All AP</SelectItem>
                <SelectItem value="H">High (H)</SelectItem>
                <SelectItem value="M">Medium (M)</SelectItem>
                <SelectItem value="L">Low (L)</SelectItem>
              </SelectContent>
            </Select>
            <Select value={filterStep} onValueChange={setFilterStep}>
              <SelectTrigger className="w-48">
                <SelectValue placeholder="Process Step" />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">All Steps</SelectItem>
                {uniqueSteps.map(step => (
                  <SelectItem key={step} value={step}>{step}</SelectItem>
                ))}
              </SelectContent>
            </Select>
            <Input
              placeholder="Search failure modes, effects, causes..."
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
              className="w-72"
            />
            {(filterAP !== 'all' || filterStep !== 'all' || searchTerm) && (
              <Button 
                variant="ghost" 
                size="sm"
                onClick={() => {
                  setFilterAP('all');
                  setFilterStep('all');
                  setSearchTerm('');
                }}
              >
                Clear filters
              </Button>
            )}
            <div className="ml-auto text-sm text-muted-foreground">
              Showing {filteredRows.length} of {summary.total} rows
            </div>
          </div>
        </CardContent>
      </Card>
      
      {/* PFMEA Table - Grouped by Step */}
      <Card>
        <CardContent className="p-0">
          <div className="overflow-x-auto">
            <Table>
              <TableHeader>
                <TableRow className="bg-muted/50">
                  <TableHead className="w-8"></TableHead>
                  <TableHead className="w-32">Step</TableHead>
                  <TableHead className="w-40">Function</TableHead>
                  <TableHead className="w-40">Failure Mode</TableHead>
                  <TableHead className="w-48">Effect</TableHead>
                  <TableHead className="w-12 text-center">S</TableHead>
                  <TableHead className="w-48">Cause</TableHead>
                  <TableHead className="w-12 text-center">O</TableHead>
                  <TableHead className="w-48">Prevention</TableHead>
                  <TableHead className="w-48">Detection</TableHead>
                  <TableHead className="w-12 text-center">D</TableHead>
                  <TableHead className="w-12 text-center">AP</TableHead>
                  <TableHead className="w-12 text-center">SC</TableHead>
                  <TableHead className="w-20">Actions</TableHead>
                </TableRow>
              </TableHeader>
              <TableBody>
                {Object.entries(rowsByStep).map(([stepName, stepRows]) => (
                  <>
                    {/* Step Group Header */}
                    <TableRow 
                      key={`step-${stepName}`} 
                      className="bg-muted/30 cursor-pointer hover:bg-muted/50"
                      onClick={() => {
                        const newExpanded = new Set(expandedRows);
                        if (newExpanded.has(stepName)) {
                          newExpanded.delete(stepName);
                        } else {
                          newExpanded.add(stepName);
                        }
                        setExpandedRows(newExpanded);
                      }}
                    >
                      <TableCell colSpan={14} className="py-2">
                        <div className="flex items-center gap-2 font-medium">
                          {expandedRows.has(stepName) ? (
                            <ChevronDown className="h-4 w-4" />
                          ) : (
                            <ChevronRight className="h-4 w-4" />
                          )}
                          {stepName}
                          <Badge variant="outline" className="ml-2">{stepRows.length} rows</Badge>
                          {stepRows.some(r => r.ap === 'H') && (
                            <Badge variant="destructive" className="ml-1">
                              {stepRows.filter(r => r.ap === 'H').length} High
                            </Badge>
                          )}
                        </div>
                      </TableCell>
                    </TableRow>
                    
                    {/* Step Rows */}
                    {expandedRows.has(stepName) && stepRows.map((row) => (
                      <TableRow 
                        key={row.id}
                        className={`
                          ${row.ap === 'H' ? 'bg-red-50/50' : ''}
                          ${row.ap === 'M' ? 'bg-yellow-50/50' : ''}
                          ${row.specialFlag ? 'border-l-4 border-l-purple-500' : ''}
                        `}
                      >
                        <TableCell></TableCell>
                        <TableCell className="text-xs text-muted-foreground">{row.stepRef}</TableCell>
                        <TableCell className="text-sm">
                          <div className="font-medium">{row.function}</div>
                          <div className="text-xs text-muted-foreground">{row.requirement}</div>
                        </TableCell>
                        <TableCell className="text-sm font-medium">{row.failureMode}</TableCell>
                        <TableCell className="text-sm">{row.effect}</TableCell>
                        <TableCell className="text-center">
                          <Badge variant={row.severity >= 9 ? 'destructive' : row.severity >= 7 ? 'outline' : 'secondary'}>
                            {row.severity}
                          </Badge>
                        </TableCell>
                        <TableCell className="text-sm">{row.cause}</TableCell>
                        <TableCell className="text-center">
                          <Badge variant={row.occurrence >= 7 ? 'destructive' : row.occurrence >= 4 ? 'outline' : 'secondary'}>
                            {row.occurrence}
                          </Badge>
                        </TableCell>
                        <TableCell className="text-xs">
                          <ul className="list-disc list-inside">
                            {row.preventionControls?.map((ctrl, i) => (
                              <li key={i}>{ctrl}</li>
                            ))}
                          </ul>
                        </TableCell>
                        <TableCell className="text-xs">
                          <ul className="list-disc list-inside">
                            {row.detectionControls?.map((ctrl, i) => (
                              <li key={i}>{ctrl}</li>
                            ))}
                          </ul>
                        </TableCell>
                        <TableCell className="text-center">
                          <Badge variant={row.detection >= 9 ? 'destructive' : row.detection >= 7 ? 'outline' : 'secondary'}>
                            {row.detection}
                          </Badge>
                        </TableCell>
                        <TableCell className="text-center">
                          <Badge variant={
                            row.ap === 'H' ? 'destructive' :
                            row.ap === 'M' ? 'default' : 'secondary'
                          }>
                            {row.ap}
                          </Badge>
                        </TableCell>
                        <TableCell className="text-center">
                          {row.csrSymbol && (
                            <TooltipProvider>
                              <Tooltip>
                                <TooltipTrigger>
                                  <span className="text-lg">{row.csrSymbol}</span>
                                </TooltipTrigger>
                                <TooltipContent>
                                  {row.csrSymbol === 'Ⓢ' ? 'Safety Critical' :
                                   row.csrSymbol === '◆' ? 'Critical' :
                                   row.csrSymbol === 'ⓒ' ? 'Significant' : 'Special'}
                                </TooltipContent>
                              </Tooltip>
                            </TooltipProvider>
                          )}
                        </TableCell>
                        <TableCell>
                          <Button 
                            variant="ghost" 
                            size="sm"
                            onClick={() => setEditingRowId(row.id)}
                          >
                            <Edit2 className="h-4 w-4" />
                          </Button>
                        </TableCell>
                      </TableRow>
                    ))}
                  </>
                ))}
              </TableBody>
            </Table>
          </div>
        </CardContent>
      </Card>
    </div>
  );
}
