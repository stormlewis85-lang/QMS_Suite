LOGIC:

2. Validate pfmeaId exists in pfmea table AND belongs to this part, throw if not found or mismatched
3. Fetch all pfmea_rows WHERE pfmea_id = pfmeaId, include parent_template_row_id
4. For each pfmea_row that has a parent_template_row_id:
   - Fetch control_template_rows WHERE source_template_row_id = parent_template_row_id
5. If no control templates found, throw error "No Control Plan templates found for this PFMEA. Ensure FMEA template rows have linked control templates."

      }

Then add to server/routes.ts:

app.post('/api/pfmeas/:id/generate-control-plan', async (req, res) => {
  const { id } = req.params;
  const { partId, type } = req.body;
  
  // Validate partId is provided
  if (!partId) {
    return res.status(400).json({ error: 'partId is required' });
  }
  
  try {
    const result = await generateControlPlan({ 
      partId, 
      pfmeaId: id,
      type: type || 'Production'
    });
    res.json(result);
  } catch (error) {
    if (error.message.includes('not found')) {
      return res.status(404).json({ error: error.message });
    }
    res.status(500).json({ error: error.message });
  }
});
