  // Add a signature (e-sign)
  app.post("/api/signatures", async (req, res) => {
    try {
      const data = req.body;
      
      // Compute content hash for the entity being signed
      let content: any = null;
      if (data.entityType === 'pfmea') {
        const pfmea = await storage.getPFMEA(data.entityId);
        const rows = await storage.getPFMEARows(data.entityId);
        content = { pfmea, rows };
      } else if (data.entityType === 'control_plan') {
        const cp = await storage.getControlPlan(data.entityId);
        const rows = await storage.getControlPlanRows(data.entityId);
        content = { controlPlan: cp, rows };
      } else if (data.entityType === 'process_def') {
        const process = await storage.getProcess(data.entityId);
        content = process;
      }

      const contentHash = storage.computeContentHash(content);
      
      const signature = await storage.createSignature({
        ...data,
        contentHash,
      });

      // Log the signature event
      await storage.logAuditEvent(
        data.entityType,
        data.entityId,
        'sign',
        data.signerUserId,
        data.signerName,
        null,
        { role: data.role, meaning: data.meaning },
        `Signed as ${data.role}`
      );

      // Check if all required signatures are complete
      const approvalStatus = await storage.checkApprovalStatus(data.entityType, data.entityId);
      
      // If complete, update document status to effective
      if (approvalStatus.complete) {
        if (data.entityType === 'pfmea') {
          await storage.updatePFMEA(data.entityId, { 
            status: 'effective',
            approvedBy: data.signerUserId,
            approvedAt: new Date(),
            effectiveFrom: new Date(),
          });
        } else if (data.entityType === 'control_plan') {
          await storage.updateControlPlan(data.entityId, { 
            status: 'effective',
            approvedBy: data.signerUserId,
            approvedAt: new Date(),
            effectiveFrom: new Date(),
          });
        } else if (data.entityType === 'process_def') {
          await storage.updateProcess(data.entityId, { 
            status: 'effective',
            effectiveFrom: new Date(),
          });
        }
      }

      res.status(201).json({ signature, approvalStatus });
    } catch (error) {
      console.error("Error creating signature:", error);
      res.status(500).json({ message: "Failed to create signature" });
    }
  });

  app.post("/api/change-packages/:id/transition", async (req, res) => {
    try {
      const { id } = req.params;
      const { newStatus, actor, actorName, note } = req.body;
      
      const existing = await storage.getChangePackage(id);
      if (!existing) {
        return res.status(404).json({ message: "Change package not found" });
      }

      // Validate transition
      const validTransitions: Record<string, string[]> = {
        'draft': ['impact_analysis', 'cancelled'],
        'impact_analysis': ['auto_review', 'draft', 'cancelled'],
        'auto_review': ['pending_signatures', 'impact_analysis', 'cancelled'],
        'pending_signatures': ['effective', 'auto_review', 'cancelled'],
        'effective': [],
        'cancelled': ['draft'],
      };

      if (!validTransitions[existing.status]?.includes(newStatus)) {
        return res.status(400).json({ 
          message: `Invalid transition from ${existing.status} to ${newStatus}` 
        });
      }

      const pkg = await storage.updateChangePackage(id, { status: newStatus });

      // Log transition
      await storage.logAuditEvent(
        'change_package',
        id,
        'status_change',
        actor,
        actorName,
        { status: existing.status },
        { status: newStatus },
        note
      );

      res.json(pkg);
    } catch (error) {
      console.error("Error transitioning change package:", error);
      res.status(500).json({ message: "Failed to transition change package" });
    }

  // Run impact analysis for change package
  app.post("/api/change-packages/:id/analyze-impact", async (req, res) => {
    try {
      const { id } = req.params;
      const items = await storage.getChangePackageItems(id);
      
      // Find all affected parts based on template changes
      const affectedPartIds = new Set<string>();
      const apChanges: { from: string; to: string; count: number }[] = [];

      for (const item of items) {
        if (item.entityType === 'fmea_template_row' || item.entityType === 'control_template_row') {
          // Find parts using this process template
          const templateRow = item.entityType === 'fmea_template_row' 
            ? await storage.getFMEATemplateRow(item.entityId)
            : await storage.getControlTemplateRow(item.entityId);
          
          if (templateRow) {
            const partMaps = await storage.getPartProcessMapsByProcess(templateRow.processDefId);
            for (const map of partMaps) {
              affectedPartIds.add(map.partId);
            }
          }

          // Track AP changes if applicable
          if (item.entityType === 'fmea_template_row' && item.fieldChanges) {
            const apChange = (item.fieldChanges as any[]).find(f => f.field === 'ap');
            if (apChange) {
              const existing = apChanges.find(c => c.from === apChange.from && c.to === apChange.to);
              if (existing) {
                existing.count++;
              } else {
                apChanges.push({ from: apChange.from, to: apChange.to, count: 1 });
              }
            }
          }
        }
      }

      // Create affected part records
      for (const partId of affectedPartIds) {
        const part = await storage.getPart(partId);
        if (part) {
          const pfmeas = await storage.getPFMEAsByPart(partId);
          const cps = await storage.getControlPlansByPart(partId);
          
          await storage.createChangePackageAffectedPart({
            changePackageId: id,
            partId,
            partNumber: part.partNumber,
            partName: part.partName,
            currentPfmeaRev: pfmeas[0]?.rev,
            currentCpRev: cps[0]?.rev,
          });
        }
      }

      // Update impact summary
      const affectedParts = await storage.getChangePackageAffectedParts(id);
      await storage.updateChangePackage(id, {
        impactSummary: {
          affectedParts: affectedParts.length,
          affectedPfmeas: affectedParts.filter(p => p.currentPfmeaRev).length,
          affectedControlPlans: affectedParts.filter(p => p.currentCpRev).length,
          apChanges,
        },
        status: 'impact_analysis',
      });

      const pkg = await storage.getChangePackage(id);
      res.json({
        package: pkg,
        affectedParts,
        apChanges,
      });
    } catch (error) {
      console.error("Error analyzing impact:", error);
      res.status(500).json({ message: "Failed to analyze impact" });
    }
  });

  // Get revision chain for a document
  app.get("/api/revisions/:entityType/:entityId", async (req, res) => {
    try {
      const { entityType, entityId } = req.params;
      const revisions: any[] = [];
      
      let currentId: string | null = entityId;
      
      while (currentId) {
        let doc: any = null;
        
        if (entityType === 'pfmea') {
          doc = await storage.getPFMEA(currentId);
        } else if (entityType === 'control_plan') {
          doc = await storage.getControlPlan(currentId);
        } else if (entityType === 'process_def') {
          doc = await storage.getProcess(currentId);
        }
        
        if (!doc) break;
        
        revisions.push({
          id: doc.id,
          rev: doc.rev,
          status: doc.status,
          effectiveFrom: doc.effectiveFrom,
          supersedesId: doc.supersedesId,
        });
        
        currentId = doc.supersedesId;
      }
      
      res.json(revisions);
    } catch (error) {
      console.error("Error fetching revisions:", error);
      res.status(500).json({ message: "Failed to fetch revision history" });
    }
  });

  // Create new revision (supersede current)
  app.post("/api/revisions/:entityType/:entityId/supersede", async (req, res) => {
    try {
      const { entityType, entityId } = req.params;
      const { newRev, changeNote, actor, actorName } = req.body;
      
      if (entityType === 'pfmea') {
        const current = await storage.getPFMEA(entityId);
        if (!current) {
          return res.status(404).json({ message: "PFMEA not found" });
        }

        // Mark current as superseded
        await storage.updatePFMEA(entityId, { status: 'superseded' });

        // Create new revision
        const rows = await storage.getPFMEARows(entityId);
        const newPfmea = await storage.createPFMEA({
          partId: current.partId,
          rev: newRev,
          status: 'draft',
          basis: current.basis,
          docNo: current.docNo,
          supersedesId: entityId,
        });

        // Copy rows to new PFMEA
        for (const row of rows) {
          await storage.createPFMEARow({
            ...row,
            id: undefined,
            pfmeaId: newPfmea.id,
          });
        }

        // Log the revision
        await storage.logAuditEvent(
          'pfmea',
          newPfmea.id,
          'create',
          actor,
          actorName,
          null,
          { rev: newRev, supersedesId: entityId },
          changeNote || `New revision ${newRev} superseding ${current.rev}`
        );

        res.status(201).json(newPfmea);
      } else if (entityType === 'control_plan') {
        const current = await storage.getControlPlan(entityId);
        if (!current) {
          return res.status(404).json({ message: "Control Plan not found" });
        }

        await storage.updateControlPlan(entityId, { status: 'superseded' });

        const rows = await storage.getControlPlanRows(entityId);
        const newCp = await storage.createControlPlan({
          partId: current.partId,
          rev: newRev,
          type: current.type,
          status: 'draft',
          docNo: current.docNo,
          supersedesId: entityId,
        });

        for (const row of rows) {
          await storage.createControlPlanRow({
            ...row,
            id: undefined,
            controlPlanId: newCp.id,
          });
        }

        await storage.logAuditEvent(
          'control_plan',
          newCp.id,
          'create',
          actor,
          actorName,
          null,
          { rev: newRev, supersedesId: entityId },
          changeNote || `New revision ${newRev} superseding ${current.rev}`
        );

        res.status(201).json(newCp);
      } else {
        res.status(400).json({ message: "Unsupported entity type for revision" });
      }
    } catch (error) {
      console.error("Error creating revision:", error);
      res.status(500).json({ message: "Failed to create new revision" });
    }
  });
