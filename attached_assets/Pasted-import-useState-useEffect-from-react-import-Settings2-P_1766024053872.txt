// Extended process type with steps
interface ProcessWithSteps extends ProcessDef {
  steps?: ProcessStep[];
}

// Status badge colors
function getStatusBadgeVariant(status: string): "default" | "secondary" | "destructive" | "outline" {
  switch (status) {
    case "effective":
      return "default";
    case "draft":
      return "secondary";
    case "review":
      return "outline";
    case "superseded":
    case "obsolete":
      return "destructive";
    default:
      return "secondary";
  }
}

function StatusBadge({ status }: { status: string }) {
  const variant = getStatusBadgeVariant(status);
  return (
    <Badge variant={variant} className="capitalize">
      {status}
    </Badge>
  );
}

// Process form schema (extended with revision validation)
const processFormSchema = insertProcessDefSchema.extend({
  createdBy: z.string().uuid().optional(),
}).omit({ createdAt: true as never });

type ProcessFormData = z.infer<typeof processFormSchema>;

// Process Form Dialog (shared by Create and Edit)
function ProcessFormDialog({
  process,
  open,
  onOpenChange,
}: {
  process?: ProcessDef;
  open: boolean;
  onOpenChange: (open: boolean) => void;
}) {
  const { toast } = useToast();
  const isEditing = !!process;

  const form = useForm<ProcessFormData>({
    resolver: zodResolver(processFormSchema),
    defaultValues: {
      name: "",
      rev: "1.0.0",
      status: "draft",
      changeNote: "",
    },
  });

  useEffect(() => {
    if (process && open) {
      form.reset({
        name: process.name,
        rev: process.rev,
        status: process.status,
        changeNote: process.changeNote || "",
        supersedesId: process.supersedesId || undefined,
      });
    } else if (!open) {
      form.reset({
        name: "",
        rev: "1.0.0",
        status: "draft",
        changeNote: "",
      });
    }
  }, [process, open, form]);

  const mutation = useMutation({
    mutationFn: async (data: ProcessFormData) => {
      const url = isEditing ? `/api/processes/${process.id}` : "/api/processes";
      const method = isEditing ? "PATCH" : "POST";
      
      // For new processes, wrap in the expected format
      const payload = isEditing ? data : { process: data, steps: [] };
      
      const res = await apiRequest(method, url, payload);
      return await res.json() as ProcessDef;
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["/api/processes"] });
      toast({
        title: isEditing ? "Process updated" : "Process created",
        description: `The process has been successfully ${isEditing ? "updated" : "created"}.`,
      });
      onOpenChange(false);
    },
    onError: (error: Error) => {
      toast({
        variant: "destructive",
        title: "Error",
        description: error.message || `Failed to ${isEditing ? "update" : "create"} process`,
      });
    },
  });

  const onSubmit = (data: ProcessFormData) => {
    mutation.mutate(data);
  };

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="sm:max-w-[550px]">
        <DialogHeader>
          <DialogTitle>{isEditing ? "Edit Process" : "Create New Process"}</DialogTitle>
          <DialogDescription>
            {isEditing
              ? "Update process definition details. Changes will be tracked."
              : "Add a new process definition to the library. You can add steps after creation."}
          </DialogDescription>
        </DialogHeader>
        <Form {...form}>
          <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
            <FormField
              control={form.control}
              name="name"
              render={({ field }) => (
                <FormItem>
                  <FormLabel>Process Name</FormLabel>
                  <FormControl>
                    <Input 
                      placeholder="e.g., Injection Molding, Ultrasonic Welding" 
                      {...field} 
                      data-testid="input-process-name"
                    />
                  </FormControl>
                  <FormDescription>
                    A descriptive name for this manufacturing process
                  </FormDescription>
                  <FormMessage />
                </FormItem>
              )}
            />

            <div className="grid grid-cols-2 gap-4">
              <FormField
                control={form.control}
                name="rev"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Revision</FormLabel>
                    <FormControl>
                      <Input 
                        placeholder="1.0.0" 
                        {...field} 
                        data-testid="input-process-rev"
                      />
                    </FormControl>
                    <FormDescription>
                      Semantic version (e.g., 1.0.0)
                    </FormDescription>
                    <FormMessage />
                  </FormItem>
                )}
              />

              <FormField
                control={form.control}
                name="status"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Status</FormLabel>
                    <Select onValueChange={field.onChange} value={field.value || "draft"}>
                      <FormControl>
                        <SelectTrigger data-testid="select-process-status">
                          <SelectValue placeholder="Select status" />
                        </SelectTrigger>
                      </FormControl>
                      <SelectContent>
                        <SelectItem value="draft">Draft</SelectItem>
                        <SelectItem value="review">Under Review</SelectItem>
                        <SelectItem value="effective">Effective</SelectItem>
                        <SelectItem value="superseded">Superseded</SelectItem>
                        <SelectItem value="obsolete">Obsolete</SelectItem>
                      </SelectContent>
                    </Select>
                    <FormMessage />
                  </FormItem>
                )}
              />
            </div>

            <FormField
              control={form.control}
              name="changeNote"
              render={({ field }) => (
                <FormItem>
                  <FormLabel>Change Notes</FormLabel>
                  <FormControl>
                    <Textarea
                      placeholder="Describe what changed in this revision..."
                      className="resize-none"
                      rows={3}
                      {...field}
                      value={field.value || ""}
                      data-testid="input-process-change-note"
                    />
                  </FormControl>
                  <FormDescription>
                    Document the reason for this revision
                  </FormDescription>
                  <FormMessage />
                </FormItem>
              )}
            />

            <DialogFooter>
              <Button 
                type="button" 
                variant="outline" 
                onClick={() => onOpenChange(false)}
              >
                Cancel
              </Button>
              <Button 
                type="submit" 
                disabled={mutation.isPending}
                data-testid="button-process-submit"
              >
                {mutation.isPending && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
                {isEditing ? "Save Changes" : "Create Process"}
              </Button>
            </DialogFooter>
          </form>
        </Form>
      </DialogContent>
    </Dialog>
  );
}

// Process Detail View Dialog
function ProcessDetailDialog({
  processId,
  open,
  onOpenChange,
}: {
  processId: string | null;
  open: boolean;
  onOpenChange: (open: boolean) => void;
}) {
  const { data: process, isLoading } = useQuery<ProcessWithSteps>({
    queryKey: ["/api/processes", processId],
    enabled: !!processId && open,
  });

  if (!processId) return null;

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="sm:max-w-[700px] max-h-[80vh] overflow-y-auto">
        <DialogHeader>
          <DialogTitle className="flex items-center gap-2">
            <Settings2 className="h-5 w-5" />
            Process Details
          </DialogTitle>
        </DialogHeader>

        {isLoading ? (
          <div className="flex items-center justify-center py-8">
            <Loader2 className="h-8 w-8 animate-spin text-muted-foreground" />
          </div>
        ) : process ? (
          <div className="space-y-6">
            {/* Header Info */}
            <div className="grid grid-cols-2 gap-4">
              <div>
                <p className="text-sm text-muted-foreground">Process Name</p>
                <p className="font-medium">{process.name}</p>
              </div>
              <div>
                <p className="text-sm text-muted-foreground">Revision</p>
                <p className="font-medium font-mono">{process.rev}</p>
              </div>
              <div>
                <p className="text-sm text-muted-foreground">Status</p>
                <StatusBadge status={process.status} />
              </div>
              <div>
                <p className="text-sm text-muted-foreground">Effective From</p>
                <p className="font-medium">
                  {process.effectiveFrom 
                    ? new Date(process.effectiveFrom).toLocaleDateString()
                    : "Not set"}
                </p>
              </div>
            </div>

            {process.changeNote && (
              <div>
                <p className="text-sm text-muted-foreground">Change Notes</p>
                <p className="text-sm mt-1 p-3 bg-muted rounded-md">{process.changeNote}</p>
              </div>
            )}

            <Separator />

            {/* Process Steps */}
            <div>
              <h3 className="font-semibold mb-3 flex items-center gap-2">
                <GitBranch className="h-4 w-4" />
                Process Steps
              </h3>
              
              {process.steps && process.steps.length > 0 ? (
                <div className="space-y-2">
                  {process.steps
                    .sort((a, b) => a.seq - b.seq)
                    .map((step, index) => (
                      <div 
                        key={step.id} 
                        className="flex items-center gap-3 p-3 border rounded-lg"
                      >
                        <div className="flex items-center justify-center w-8 h-8 rounded-full bg-primary/10 text-primary font-medium text-sm">
                          {step.seq}
                        </div>
                        <div className="flex-1">
                          <p className="font-medium">{step.name}</p>
                          <p className="text-sm text-muted-foreground">{step.area}</p>
                        </div>
                        {step.stepType !== "operation" && (
                          <Badge variant="outline" className="text-xs">
                            {step.stepType === "group" ? "Group" : "Subprocess"}
                          </Badge>
                        )}
                        {step.equipment && step.equipment.length > 0 && (
                          <Badge variant="secondary" className="text-xs">
                            {step.equipment.length} Equipment
                          </Badge>
                        )}
                        {index < (process.steps?.length || 0) - 1 && (
                          <ChevronRight className="h-4 w-4 text-muted-foreground" />
                        )}
                      </div>
                    ))}
                </div>
              ) : (
                <div className="text-center py-8 text-muted-foreground border rounded-lg border-dashed">
                  <FileText className="h-8 w-8 mx-auto mb-2 opacity-50" />
                  <p>No steps defined yet</p>
                  <p className="text-sm">Add steps to this process from the edit view</p>
                </div>
              )}
            </div>

            {/* Metadata */}
            <Separator />
            <div className="text-xs text-muted-foreground">
              <p>Created: {new Date(process.createdAt).toLocaleString()}</p>
              <p>Process ID: {process.id}</p>
            </div>
          </div>
        ) : (
          <div className="text-center py-8 text-muted-foreground">
            Process not found
          </div>
        )}
      </DialogContent>
    </Dialog>
  );
}

// Delete Confirmation Dialog
function DeleteConfirmDialog({
  process,
  open,
  onOpenChange,
}: {
  process: ProcessDef | null;
  open: boolean;
  onOpenChange: (open: boolean) => void;
}) {
  const { toast } = useToast();

  const mutation = useMutation({
    mutationFn: async () => {
      if (!process) return;
      await apiRequest("DELETE", `/api/processes/${process.id}`);
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["/api/processes"] });
      toast({
        title: "Process deleted",
        description: "The process has been permanently removed.",
      });
      onOpenChange(false);
    },
    onError: (error: Error) => {
      toast({
        variant: "destructive",
        title: "Error",
        description: error.message || "Failed to delete process",
      });
    },
  });

  if (!process) return null;

  return (
    <AlertDialog open={open} onOpenChange={onOpenChange}>
      <AlertDialogContent>
        <AlertDialogHeader>
          <AlertDialogTitle className="flex items-center gap-2">
            <AlertTriangle className="h-5 w-5 text-destructive" />
            Delete Process
          </AlertDialogTitle>
          <AlertDialogDescription>
            Are you sure you want to delete <strong>"{process.name}"</strong> (Rev {process.rev})?
            <br /><br />
            This action cannot be undone. All associated steps, FMEA template rows, 
            and control plan template rows will also be deleted.
          </AlertDialogDescription>
        </AlertDialogHeader>
        <AlertDialogFooter>
          <AlertDialogCancel>Cancel</AlertDialogCancel>
          <AlertDialogAction
            onClick={() => mutation.mutate()}
            disabled={mutation.isPending}
            className="bg-destructive text-destructive-foreground hover:bg-destructive/90"
          >
            {mutation.isPending && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
            Delete Process
          </AlertDialogAction>
        </AlertDialogFooter>
      </AlertDialogContent>
    </AlertDialog>
  );
}

// Main Processes Page Component
export default function Processes() {
  const [searchQuery, setSearchQuery] = useState("");
  const [statusFilter, setStatusFilter] = useState<string>("all");
  
  // Dialog states
  const [createDialogOpen, setCreateDialogOpen] = useState(false);
  const [editDialogOpen, setEditDialogOpen] = useState(false);
  const [viewDialogOpen, setViewDialogOpen] = useState(false);
  const [deleteDialogOpen, setDeleteDialogOpen] = useState(false);
  
  // Selected process for dialogs
  const [selectedProcess, setSelectedProcess] = useState<ProcessDef | null>(null);

  // Fetch all processes
  const { data: processes, isLoading, error } = useQuery<ProcessDef[]>({
    queryKey: ["/api/processes"],
  });

  // Filter processes
  const filteredProcesses = processes?.filter((process) => {
    const matchesSearch =
      process.name.toLowerCase().includes(searchQuery.toLowerCase()) ||
      process.rev.toLowerCase().includes(searchQuery.toLowerCase());
    const matchesStatus = statusFilter === "all" || process.status === statusFilter;
    return matchesSearch && matchesStatus;
  }) || [];

  // Group processes by status for summary
  const statusCounts = {
    all: processes?.length || 0,
    effective: processes?.filter((p) => p.status === "effective").length || 0,
    draft: processes?.filter((p) => p.status === "draft").length || 0,
    review: processes?.filter((p) => p.status === "review").length || 0,
    superseded: processes?.filter((p) => p.status === "superseded").length || 0,
    obsolete: processes?.filter((p) => p.status === "obsolete").length || 0,
  };

  // Handlers
  const handleView = (process: ProcessDef) => {
    setSelectedProcess(process);
    setViewDialogOpen(true);
  };

  const handleEdit = (process: ProcessDef) => {
    setSelectedProcess(process);
    setEditDialogOpen(true);
  };

  const handleDelete = (process: ProcessDef) => {
    setSelectedProcess(process);
    setDeleteDialogOpen(true);
  };

  const handleDuplicate = async (process: ProcessDef) => {
    try {
      const newProcess = {
        process: {
          name: `${process.name} (Copy)`,
          rev: "1.0.0",
          status: "draft" as const,
          changeNote: `Duplicated from ${process.name} Rev ${process.rev}`,
        },
        steps: [],
      };
      
      await apiRequest("POST", "/api/processes", newProcess);
      queryClient.invalidateQueries({ queryKey: ["/api/processes"] });
    } catch (error) {
      console.error("Failed to duplicate process:", error);
    }
  };

  return (
    <div className="space-y-6">
      {/* Page Header */}
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-3xl font-bold tracking-tight">Process Library</h1>
          <p className="text-muted-foreground">
            Manage process definitions, steps, and FMEA templates
          </p>
        </div>
        <Button onClick={() => setCreateDialogOpen(true)} data-testid="button-create-process">
          <Plus className="mr-2 h-4 w-4" />
          Create Process
        </Button>
      </div>

      {/* Stats Cards */}
      <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-6 gap-4">
        {[
          { label: "All", value: statusCounts.all, filter: "all" },
          { label: "Effective", value: statusCounts.effective, filter: "effective" },
          { label: "Draft", value: statusCounts.draft, filter: "draft" },
          { label: "Review", value: statusCounts.review, filter: "review" },
          { label: "Superseded", value: statusCounts.superseded, filter: "superseded" },
          { label: "Obsolete", value: statusCounts.obsolete, filter: "obsolete" },
        ].map((stat) => (
          <Card
            key={stat.filter}
            className={`cursor-pointer transition-colors hover:bg-accent ${
              statusFilter === stat.filter ? "ring-2 ring-primary" : ""
            }`}
            onClick={() => setStatusFilter(stat.filter)}
          >
            <CardContent className="p-4">
              <p className="text-2xl font-bold">{stat.value}</p>
              <p className="text-sm text-muted-foreground">{stat.label}</p>
            </CardContent>
          </Card>
        ))}
      </div>

      {/* Search and Filters */}
      <Card>
        <CardHeader className="pb-3">
          <div className="flex items-center gap-4">
            <div className="relative flex-1 max-w-sm">
              <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
              <Input
                placeholder="Search processes..."
                value={searchQuery}
                onChange={(e) => setSearchQuery(e.target.value)}
                className="pl-9"
                data-testid="input-search-processes"
              />
              {searchQuery && (
                <Button
                  variant="ghost"
                  size="sm"
                  className="absolute right-1 top-1/2 -translate-y-1/2 h-7 w-7 p-0"
                  onClick={() => setSearchQuery("")}
                >
                  <X className="h-4 w-4" />
                </Button>
              )}
            </div>
            <Select value={statusFilter} onValueChange={setStatusFilter}>
              <SelectTrigger className="w-[150px]" data-testid="select-status-filter">
                <SelectValue placeholder="Filter by status" />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">All Status</SelectItem>
                <SelectItem value="draft">Draft</SelectItem>
                <SelectItem value="review">Review</SelectItem>
                <SelectItem value="effective">Effective</SelectItem>
                <SelectItem value="superseded">Superseded</SelectItem>
                <SelectItem value="obsolete">Obsolete</SelectItem>
              </SelectContent>
            </Select>
          </div>
        </CardHeader>
        <CardContent>
          {isLoading ? (
            <div className="flex items-center justify-center py-8">
              <Loader2 className="h-8 w-8 animate-spin text-muted-foreground" />
            </div>
          ) : error ? (
            <div className="text-center py-8 text-destructive">
              <AlertTriangle className="h-8 w-8 mx-auto mb-2" />
              <p>Failed to load processes</p>
              <p className="text-sm text-muted-foreground mt-1">
                {error instanceof Error ? error.message : "Unknown error"}
              </p>
            </div>
          ) : filteredProcesses.length === 0 ? (
            <div className="text-center py-8 text-muted-foreground">
              <Settings2 className="h-12 w-12 mx-auto mb-4 opacity-50" />
              {searchQuery || statusFilter !== "all" ? (
                <>
                  <p className="font-medium">No processes match your filters</p>
                  <p className="text-sm mt-1">Try adjusting your search or filter criteria</p>
                  <Button
                    variant="outline"
                    className="mt-4"
                    onClick={() => {
                      setSearchQuery("");
                      setStatusFilter("all");
                    }}
                  >
                    Clear Filters
                  </Button>
                </>
              ) : (
                <>
                  <p className="font-medium">No processes yet</p>
                  <p className="text-sm mt-1">Create your first process definition to get started</p>
                  <Button className="mt-4" onClick={() => setCreateDialogOpen(true)}>
                    <Plus className="mr-2 h-4 w-4" />
                    Create Process
                  </Button>
                </>
              )}
            </div>
          ) : (
            <Table>
              <TableHeader>
                <TableRow>
                  <TableHead>Process Name</TableHead>
                  <TableHead>Revision</TableHead>
                  <TableHead>Status</TableHead>
                  <TableHead>Effective From</TableHead>
                  <TableHead>Created</TableHead>
                  <TableHead className="w-[100px]">Actions</TableHead>
                </TableRow>
              </TableHeader>
              <TableBody>
                {filteredProcesses.map((process) => (
                  <TableRow key={process.id} data-testid={`process-row-${process.id}`}>
                    <TableCell className="font-medium">
                      <div className="flex items-center gap-2">
                        <Settings2 className="h-4 w-4 text-muted-foreground" />
                        {process.name}
                      </div>
                    </TableCell>
                    <TableCell>
                      <code className="text-sm bg-muted px-2 py-1 rounded">
                        {process.rev}
                      </code>
                    </TableCell>
                    <TableCell>
                      <StatusBadge status={process.status} />
                    </TableCell>
                    <TableCell>
                      {process.effectiveFrom
                        ? new Date(process.effectiveFrom).toLocaleDateString()
                        : "—"}
                    </TableCell>
                    <TableCell className="text-muted-foreground text-sm">
                      {new Date(process.createdAt).toLocaleDateString()}
                    </TableCell>
                    <TableCell>
                      <DropdownMenu>
                        <DropdownMenuTrigger asChild>
                          <Button variant="ghost" size="sm" data-testid={`button-actions-${process.id}`}>
                            •••
                          </Button>
                        </DropdownMenuTrigger>
                        <DropdownMenuContent align="end">
                          <DropdownMenuItem onClick={() => handleView(process)}>
                            <Eye className="mr-2 h-4 w-4" />
                            View Details
                          </DropdownMenuItem>
                          <DropdownMenuItem onClick={() => handleEdit(process)}>
                            <Pencil className="mr-2 h-4 w-4" />
                            Edit
                          </DropdownMenuItem>
                          <DropdownMenuItem onClick={() => handleDuplicate(process)}>
                            <Copy className="mr-2 h-4 w-4" />
                            Duplicate
                          </DropdownMenuItem>
                          <DropdownMenuSeparator />
                          <DropdownMenuItem 
                            onClick={() => handleDelete(process)}
                            className="text-destructive focus:text-destructive"
                          >
                            <Trash2 className="mr-2 h-4 w-4" />
                            Delete
                          </DropdownMenuItem>
                        </DropdownMenuContent>
                      </DropdownMenu>
                    </TableCell>
                  </TableRow>
                ))}
              </TableBody>
            </Table>
          )}
        </CardContent>
      </Card>

      {/* Dialogs */}
      <ProcessFormDialog
        open={createDialogOpen}
        onOpenChange={setCreateDialogOpen}
      />
      
      <ProcessFormDialog
        process={selectedProcess || undefined}
        open={editDialogOpen}
        onOpenChange={setEditDialogOpen}
      />
      
      <ProcessDetailDialog
        processId={selectedProcess?.id || null}
        open={viewDialogOpen}
        onOpenChange={setViewDialogOpen}
      />
      
      <DeleteConfirmDialog
        process={selectedProcess}
        open={deleteDialogOpen}
        onOpenChange={setDeleteDialogOpen}
      />
    </div>
  );
}
