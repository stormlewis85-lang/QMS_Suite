// packages/api/src/routers/failure-modes-library.ts

import { z } from 'zod';
import { router, publicProcedure, protectedProcedure } from '../trpc.js';
import { db, failureModesLibrary } from '@pfmea/db';
import { eq, like, and, inArray } from 'drizzle-orm';

export const failureModesRouter = router({
  list: publicProcedure
    .input(z.object({
      category: z.enum(['dimensional', 'visual', 'functional', 'assembly', 'material', 'process', 'contamination', 'environmental']).optional(),
      search: z.string().optional(),
      tags: z.array(z.string()).optional(),
      applicableProcess: z.string().optional(),
    }))
    .query(async ({ input }) => {
      const conditions = [];
      
      if (input.category) {
        conditions.push(eq(failureModesLibrary.category, input.category));
      }
      
      if (input.search) {
        conditions.push(
          like(failureModesLibrary.failureMode, `%${input.search}%`)
        );
      }
      
      const where = conditions.length > 0 ? and(...conditions) : undefined;
      
      let modes = await db.query.failureModesLibrary.findMany({
        where,
        orderBy: (fm, { desc }) => [desc(fm.lastUsed), desc(fm.createdAt)],
      });
      
      // Filter by tags (JSON field)
      if (input.tags && input.tags.length > 0) {
        modes = modes.filter(fm => 
          input.tags!.some(tag => (fm.tags as string[]).includes(tag))
        );
      }
      
      // Filter by applicable process (JSON field)
      if (input.applicableProcess) {
        modes = modes.filter(fm => 
          (fm.applicableProcesses as string[]).includes(input.applicableProcess!)
        );
      }
      
      return modes;
    }),

  getById: publicProcedure
    .input(z.object({ id: z.string() }))
    .query(async ({ input }) => {
      return db.query.failureModesLibrary.findFirst({
        where: eq(failureModesLibrary.id, input.id),
      });
    }),

  create: protectedProcedure
    .input(z.object({
      category: z.enum(['dimensional', 'visual', 'functional', 'assembly', 'material', 'process', 'contamination', 'environmental']),
      failureMode: z.string().min(3),
      genericEffect: z.string().min(3),
      typicalCauses: z.array(z.string()),
      applicableProcesses: z.array(z.string()).optional(),
      defaultSeverity: z.number().min(1).max(10).optional(),
      defaultOccurrence: z.number().min(1).max(10).optional(),
      tags: z.array(z.string()).optional(),
      industryStandard: z.string().optional(),
    }))
    .mutation(async ({ input, ctx }) => {
      const [newMode] = await db.insert(failureModesLibrary).values({
        ...input,
        createdBy: ctx.user.id,
      }).returning();
      
      return newMode;
    }),

  update: protectedProcedure
    .input(z.object({
      id: z.string(),
      category: z.enum(['dimensional', 'visual', 'functional', 'assembly', 'material', 'process', 'contamination', 'environmental']).optional(),
      failureMode: z.string().optional(),
      genericEffect: z.string().optional(),
      typicalCauses: z.array(z.string()).optional(),
      applicableProcesses: z.array(z.string()).optional(),
      defaultSeverity: z.number().min(1).max(10).optional(),
      defaultOccurrence: z.number().min(1).max(10).optional(),
      tags: z.array(z.string()).optional(),
      status: z.enum(['active', 'deprecated']).optional(),
    }))
    .mutation(async ({ input }) => {
      const { id, ...updates } = input;
      
      await db.update(failureModesLibrary)
        .set(updates)
        .where(eq(failureModesLibrary.id, id));
      
      return { success: true };
    }),

  markUsed: publicProcedure
    .input(z.object({ id: z.string() }))
    .mutation(async ({ input }) => {
      await db.update(failureModesLibrary)
        .set({ lastUsed: new Date() })
        .where(eq(failureModesLibrary.id, input.id));
      
      return { success: true };
    }),

  getCategories: publicProcedure.query(() => {
    return [
      { value: 'dimensional', label: 'Dimensional', icon: '📏' },
      { value: 'visual', label: 'Visual/Cosmetic', icon: '👁️' },
      { value: 'functional', label: 'Functional', icon: '⚙️' },
      { value: 'assembly', label: 'Assembly', icon: '🔧' },
      { value: 'material', label: 'Material', icon: '🧱' },
      { value: 'process', label: 'Process', icon: '🏭' },
      { value: 'contamination', label: 'Contamination', icon: '⚠️' },
      { value: 'environmental', label: 'Environmental', icon: '🌡️' },
    ];
  }),

  getUsageStats: publicProcedure.query(async () => {
    const modes = await db.query.failureModesLibrary.findMany();
    
    const categoryStats = modes.reduce((acc, mode) => {
      acc[mode.category] = (acc[mode.category] || 0) + 1;
      return acc;
    }, {} as Record<string, number>);
    
    const recentlyUsed = modes
      .filter(m => m.lastUsed)
      .sort((a, b) => (b.lastUsed?.getTime() || 0) - (a.lastUsed?.getTime() || 0))
      .slice(0, 10);
    
    return {
      totalModes: modes.length,
      categoryBreakdown: categoryStats,
      recentlyUsed,
    };
  }),
});
