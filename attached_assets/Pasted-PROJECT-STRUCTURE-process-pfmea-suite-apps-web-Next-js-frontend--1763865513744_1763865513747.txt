Dockerfile
dockerfileFROM node:20-alpine AS base
RUN corepack enable pnpm

FROM base AS deps
WORKDIR /app
COPY package.json pnpm-lock.yaml pnpm-workspace.yaml turbo.json ./
COPY apps/web/package.json ./apps/web/
COPY packages/*/package.json ./packages/
RUN pnpm install --frozen-lockfile

FROM base AS builder
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY . .
RUN pnpm db:migrate
RUN pnpm build

FROM base AS runner
WORKDIR /app
ENV NODE_ENV=production
COPY --from=builder /app/apps/web/.next/standalone ./
COPY --from=builder /app/apps/web/.next/static ./apps/web/.next/static
COPY --from=builder /app/apps/web/public ./apps/web/public
EXPOSE 3000

typescriptimport { pgTable, text, integer, jsonb, timestamp, boolean, uuid, pgEnum, index, uniqueIndex } from 'drizzle-orm/pg-core';
import { relations } from 'drizzle-orm';

// Enums
export const statusEnum = pgEnum('status', ['draft', 'review', 'effective', 'superseded', 'obsolete']);
export const changeStatusEnum = pgEnum('change_status', ['draft', 'impact_analysis', 'auto_review', 'pending_signatures', 'effective', 'cancelled']);
export const roleEnum = pgEnum('role', ['library_maintainer', 'part_engineer', 'qe', 'process_owner', 'quality_manager', 'auditor']);

// Process Library Tables
export const processDef = pgTable('process_def', {
  id: uuid('id').primaryKey().defaultRandom(),
  name: text('name').notNull(),
  rev: text('rev').notNull(),
  status: statusEnum('status').notNull().default('draft'),
  effectiveFrom: timestamp('effective_from'),
  supersedesId: uuid('supersedes_id').references(() => processDef.id),
  changeNote: text('change_note'),
  createdBy: uuid('created_by').notNull(),
  createdAt: timestamp('created_at').notNull().defaultNow(),
}, (table) => ({
  nameRevIdx: uniqueIndex('process_def_name_rev_idx').on(table.name, table.rev),
  statusIdx: index('process_def_status_idx').on(table.status),
}));

export const processStep = pgTable('process_step', {
  id: uuid('id').primaryKey().defaultRandom(),
  processDefId: uuid('process_def_id').notNull().references(() => processDef.id, { onDelete: 'cascade' }),
  seq: integer('seq').notNull(),
  name: text('name').notNull(),
  area: text('area').notNull(),
  equipment: jsonb('equipment').$type<{ name: string; model?: string; settings?: Record<string, any> }[]>(),
  branchTo: text('branch_to'),
  reworkTo: text('rework_to'),
}, (table) => ({
  processDefSeqIdx: uniqueIndex('process_step_def_seq_idx').on(table.processDefId, table.seq),
}));

export const fmeaTemplateRow = pgTable('fmea_template_row', {
  id: uuid('id').primaryKey().defaultRandom(),
  processDefId: uuid('process_def_id').notNull().references(() => processDef.id, { onDelete: 'cascade' }),
  stepId: uuid('step_id').notNull().references(() => processStep.id, { onDelete: 'cascade' }),
  function: text('function').notNull(),
  requirement: text('requirement').notNull(),
  failureMode: text('failure_mode').notNull(),
  effect: text('effect').notNull(),
  severity: integer('severity').notNull(),
  cause: text('cause').notNull(),
  occurrence: integer('occurrence').notNull(),
  preventionControls: jsonb('prevention_controls').$type<string[]>().notNull().default([]),
  detectionControls: jsonb('detection_controls').$type<string[]>().notNull().default([]),
  detection: integer('detection').notNull(),

typescriptimport type { AP, APCalculationResult } from './types.js';

interface RatingScale {
  rating: number;
  description: string;
  criteria: string;
}

interface APMatrix {
  S: RatingScale[];
  O: RatingScale[];
  D: RatingScale[];
}

export class APCalculator {
  constructor(private matrix: APMatrix) {}

  calculate(S: number, O: number, D: number): APCalculationResult {
    // AIAG-VDA 2019 Action Priority Logic
    
    // HIGH: S ≥ 9 regardless of O or D
    if (S >= 9) {
      return { ap: 'H', reason: 'Severity ≥ 9 (Safety critical)' };
    }

    // HIGH: S = 7-8 AND O ≥ 7
    if (S >= 7 && S <= 8 && O >= 7) {
      return { ap: 'H', reason: 'Severity 7-8 with high occurrence ≥ 7' };
    }

    // HIGH: S = 7-8 AND D ≥ 7
    if (S >= 7 && S <= 8 && D >= 7) {
      return { ap: 'H', reason: 'Severity 7-8 with poor detection ≥ 7' };
    }

    // HIGH: Detection-only (D ≥ 9) regardless of S/O
    if (D >= 9) {
      return { ap: 'H', reason: 'Detection ≥ 9 (no control)' };
    }

    // MEDIUM: S = 5-6 AND (O ≥ 7 OR D ≥ 7)
    if (S >= 5 && S <= 6 && (O >= 7 || D >= 7)) {
      return { ap: 'M', reason: 'Moderate severity with high O or D' };
    }

    // MEDIUM: S = 7-8 AND O = 4-6 AND D = 4-6
    if (S >= 7 && S <= 8 && O >= 4 && O <= 6 && D >= 4 && D <= 6) {
      return { ap: 'M', reason: 'High severity with moderate O and D' };
    }

    // LOW: All other combinations
    return { ap: 'L', reason: 'Low risk - effective controls in place' };
  }

  recompute(rows: Array<{ severity: number; occurrence: number; detection: number }>): Array<APCalculationResult> {
    return rows.map(({ severity, occurrence, detection }) => 
      this.calculate(severity, occurrence, detection)
    );
  }
}

typescriptimport { db, processStep, fmeaTemplateRow, controlTemplateRow } from '@pfmea/db';
import { eq, inArray } from 'drizzle-orm';

interface ProcessSelection {
  processDefId: string;
  rev: string;
}

export class PFDGenerator {
  async generate(selections: ProcessSelection[]) {
    const allSteps = [];
    
    for (const selection of selections) {
      const steps = await db.query.processStep.findMany({
        where: eq(processStep.processDefId, selection.processDefId),
        orderBy: (step, { asc }) => [asc(step.seq)],
      });
      allSteps.push(...steps);
    }

    // Generate Mermaid
    const mermaid = this.generateMermaid(allSteps);
    
    // Generate JSON structure
    const json = {
      steps: allSteps.map((step) => ({
        id: step.id,
        seq: step.seq,
        name: step.name,
        area: step.area,
        equipment: step.equipment,
      })),
    };

    return { mermaid, json };
  }

  private generateMermaid(steps: any[]): string {
    let mermaid = 'graph LR\n';
    mermaid += '  Start([Start]) --> ';
    
    steps.forEach((step, idx) => {
      const nodeId = `S${step.seq}`;
      const label = `${step.seq}: ${step.name}<br/>${step.area}`;
      
      if (idx === 0) {
        mermaid += `${nodeId}["${label}"]\n`;
      } else {
        const prevNodeId = `S${steps[idx - 1].seq}`;
        mermaid += `  ${prevNodeId} --> ${nodeId}["${label}"]\n`;
      }
    });
    
    const lastNodeId = `S${steps[steps.length - 1].seq}`;
    mermaid += `  ${lastNodeId} --> End([End])\n`;
    
    return mermaid;
  }
}

typescriptimport { db, pfd, pfmea, pfmeaRow, controlPlan, controlPlanRow } from '@pfmea/db';
import { eq } from 'drizzle-orm';
import type { AutoReviewFinding } from '@pfmea/shared';
import { APCalculator } from '@pfmea/shared';

export class AutoReviewService {
  constructor(private apCalculator: APCalculator) {}

  async runReview(pfmeaId: string, controlPlanId: string): Promise<AutoReviewFinding[]> {
    const findings: AutoReviewFinding[] = [];

    // Get data
    const pfmeaData = await db.query.pfmea.findFirst({
      where: eq(pfmea.id, pfmeaId),
      with: {
        part: {
          with: {
            pfds: { limit: 1, orderBy: (pfd, { desc }) => [desc(pfd.createdAt)] },
          },
        },
        rows: true,
      },
    });

    const cpData = await db.query.controlPlan.findFirst({
      where: eq(controlPlan.id, controlPlanId),
      with: {
        rows: true,
      },
    });

    if (!pfmeaData || !cpData) {
      findings.push({
        level: 'error',
        category: 'coverage',
        message: 'Missing PFMEA or Control Plan data',
      });
      return findings;
    }

    // 1. PFD→PFMEA Coverage
    const pfdData = pfmeaData.part.pfds[0];
    if (pfdData) {
      const pfdSteps = (pfdData.json as any).steps || [];
      const pfmeaStepRefs = new Set(pfmeaData.rows.map(r => r.stepRef));
      
      for (const step of pfdSteps) {
        if (!pfmeaStepRefs.has(step.name)) {
          findings.push({
            level: 'error',
            category: 'coverage',
            message: `PFD step "${step.name}" has no PFMEA coverage`,
            ref: step.id,
          });
        }
      }
    }

    // 2. PFMEA→CP Coverage (AP=HIGH must have CP row or waiver)
    const highAPRows = pfmeaData.rows.filter(r => r.ap === 'H');
    for (const fmeaRow of highAPRows) {
      const hasCPRow = cpData.rows.some(cp => cp.sourcePfmeaRowId === fmeaRow.id);
      if (!hasCPRow) {
        findings.push({
          level: 'error',
          category: 'coverage',
          message: `High AP PFMEA row "${fmeaRow.failureMode}" has no Control Plan coverage`,
          ref: fmeaRow.id,
        });
      }
    }

    // 3. AP Recompute Check
    for (const row of pfmeaData.rows) {
      const computed = this.apCalculator.calculate(row.severity, row.occurrence, row.detection);
      if (computed.ap !== row.ap) {
        findings.push({
          level: 'warning',
          category: 'effectiveness',
          message: `PFMEA row "${row.failureMode}" AP mismatch: stored=${row.ap}, computed=${computed.ap}`,
          ref: row.id,
        });
      }
    }

    // 4. Detection-only with S≥7 requires prevention
    const detectionOnlyHighSeverity = pfmeaData.rows.filter(
      r => r.severity >= 7 && (!r.preventionControls || r.preventionControls.length === 0)
    );
    for (const row of detectionOnlyHighSeverity) {
      findings.push({
        level: 'error',
        category: 'effectiveness',
        message: `PFMEA row "${row.failureMode}" has S≥7 but no prevention controls`,
        ref: row.id,
      });
    }

    // 5. CSR with capability targets
    const csrRows = cpData.rows.filter(r => r.csrSymbol);
    for (const row of csrRows) {
      if (!row.acceptanceCriteria?.includes('Cpk')) {
        findings.push({
          level: 'warning',
          category: 'effectiveness',
          message: `CSR characteristic "${row.characteristicName}" missing Cpk target`,
          ref: row.id,
        });
      }
      if (!row.reactionPlan) {
        findings.push({
          level: 'error',
          category: 'effectiveness',
          message: `CSR characteristic "${row.characteristicName}" missing reaction plan`,
          ref: row.id,
        });
      }
    }

    // 6. Document control
    if (!pfmeaData.docNo) {
      findings.push({
        level: 'error',
        category: 'document_control',
        message: 'PFMEA missing document number',
      });
    }

    if (!cpData.docNo) {
      findings.push({
        level: 'error',
        category: 'document_control',
        message: 'Control Plan missing document number',
      });
    }

    return findings;
  }
}

  generate: publicProcedure
    .input(z.object({
      partId: z.string(),

        processDefId: z.string(),
        rev: z.string(),

## Auto-Review Rules

The system validates:

1. **Coverage**
   - Every PFD step has ≥1 PFMEA row
   - Every AP=HIGH PFMEA row has ≥1 Control Plan row or waiver
   - No orphaned references

2. **Effectiveness**
   - AP matches S×O×D per AIAG-VDA 2019
   - Detection-only (S≥7) requires prevention
   - CSR characteristics have capability targets + reaction plans
   - MSA/Calibration status OK for gages

3. **Document Control**
   - Doc No/Rev/Effective present
   - QR code resolves to snapshot
   - Content hash verification

## Change Package Workflow

1. **Draft** → Modify templates
2. **Impact Analysis** → Compute affected parts, AP deltas, redlines
3. **Auto-Review** → Validate compliance
4. **Pending Signatures** → Route for approval
5. **Effective** → Propagate changes (adopt vs keep override)
